SELECT SUM(CASE automobiles        WHEN 'Y' THEN 1 ELSE 0 END) automobiles       ,SUM(CASE buses              WHEN 'Y' THEN 1 ELSE 0 END) buses             ,SUM(CASE taxis              WHEN 'Y' THEN 1 ELSE 0 END) taxis             ,SUM(CASE carpools           WHEN 'Y' THEN 1 ELSE 0 END) carpools          ,SUM(CASE pedestrians        WHEN 'Y' THEN 1 ELSE 0 END) pedestrians       ,SUM(CASE trucks             WHEN 'Y' THEN 1 ELSE 0 END) trucks            ,SUM(CASE deliveries         WHEN 'Y' THEN 1 ELSE 0 END) deliveries        ,SUM(CASE emergency_vehicles WHEN 'Y' THEN 1 ELSE 0 END) emergency_vehicles,SUM(CASE through_traffic    WHEN 'Y' THEN 1 ELSE 0 END) through_traffic   ,SUM(CASE motorcycles        WHEN 'Y' THEN 1 ELSE 0 END) motorcycles       FROM (SELECT rdf_access.* FROM rdf_access, rdf_condition WHERE rdf_condition.access_id=rdf_access.access_id AND condition_type=23) a;
