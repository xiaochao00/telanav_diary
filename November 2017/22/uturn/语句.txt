(select feature_id,rfn.name as admin_name from rdf_feature_name rfn, rdf_feature_names rfns where rfn.name_id=rfns.name_id and name_type='B' and rfn.name = 'Reynoldsburg') b

自动名称去重的 得到 admin_name,admin_place_id,admin_type
select admin_place_id,admin_type,admin_name
from 
( select a2.feature_id,a2.admin_name from 
(
    select feature_id,min(language_code) as language_code from rdf_feature_name rfn,rdf_feature_names rfns where rfn.name_id=rfns.name_id and name_type='B' and rfns.feature_id
in (select admin_place_id from rdf_admin_attribute where admin_wide_regulations=1) 
group by feature_id 
) a1
    left join (
        select  feature_id,rfn.language_code,rfn.name as admin_name from rdf_feature_name rfn, rdf_feature_names rfns 
 where rfn.name_id=rfns.name_id and name_type='B' 
    and rfns.feature_id in (select admin_place_id from rdf_admin_attribute where admin_wide_regulations=1 )
              ) a2 on a1.feature_id=a2.feature_id and a1.language_code=a2.language_code
 ) a3 left join rdf_admin_place rap on rap.admin_place_id=a3.feature_id
